import { z } from "zod";
import OpenAI from "openai";

// Validation schema
const qualifySchema = z.object({
  patient_name: z.string().optional().default("Unknown"),
  dx: z.string().optional().default(""),
  hx: z.string().optional().default(""),
  rx: z.string().optional().default(""),
  age: z.union([z.number(), z.null()]).optional().default(null),
  gender: z.string().optional().default("Unknown"),
});

app.post("/api/ai/qualify-ancillaries", async (req, res) => {
  try {
    const validation = qualifySchema.safeParse(req.body);
    if (!validation.success) {
      return res.status(400).json({ ok: false, error: validation.error.message });
    }
    const { patient_name, dx, hx, rx, age, gender } = validation.data;
    if (!dx.trim() && !hx.trim() && !rx.trim()) {
      return res.status(400).json({ ok: false, error: "At least one of dx, hx, or rx is required" });
    }

    const openai = new OpenAI({
      apiKey: process.env.OPENAI_API_KEY,       // or your env var name
      baseURL: process.env.OPENAI_BASE_URL,     // optional, remove if using standard OpenAI
    });

    const ancillaryList = [
      { name: "BrainWave", desc: "EEG/neurocognitive testing for cognitive, neurological, mood disorders" },
      { name: "VitalWave", desc: "ANS/autonomic nervous system testing for cardiac, neuropathy, dysautonomia" },
      { name: "PGx", desc: "Pharmacogenomic testing for medication metabolism/interactions" },
      { name: "FibroScan", desc: "Liver fibrosis/steatosis assessment for fatty liver, hepatitis, NAFLD" },
      { name: "Bilateral Carotid Duplex (93880)", desc: "Carotid artery duplex ultrasound for stroke risk, hypertension, atherosclerosis, carotid stenosis" },
      { name: "Echocardiogram (93306)", desc: "Transthoracic echocardiogram for cardiac function, valve disease, heart failure, hypertension" },
      { name: "Stress Echo", desc: "Stress echocardiogram for exertional symptoms, CAD evaluation, chest pain, dyspnea" },
      { name: "Lower Extremity Venous Duplex (93971)", desc: "Lower extremity venous duplex for DVT, venous insufficiency, leg edema, varicose veins" },
      { name: "Upper Extremity Venous Duplex (93970)", desc: "Upper extremity venous duplex for DVT, arm swelling, thrombosis" },
      { name: "Lower Extremity Arterial Duplex", desc: "Lower extremity arterial duplex for PAD, claudication, arterial insufficiency" },
      { name: "Upper Extremity Arterial Duplex", desc: "Upper extremity arterial duplex for arterial insufficiency, Raynaud's" },
      { name: "AAA Ultrasound", desc: "Abdominal aortic aneurysm screening" },
      { name: "Renal Artery Duplex", desc: "Renal artery duplex for renovascular hypertension, kidney disease" },
      { name: "IVC Ultrasound", desc: "Inferior vena cava ultrasound for fluid status, heart failure" },
      { name: "Liver Ultrasound", desc: "Liver/hepatic ultrasound for fatty liver, hepatomegaly" },
      { name: "Abdominal Aorta Ultrasound", desc: "Abdominal/mesenteric aorta duplex for aortic disease" },
    ];

    const response = await openai.chat.completions.create({
      model: "gpt-4o-mini",
      messages: [
        {
          role: "system",
          content: `You are a clinical ancillary qualification specialist. Your job is to analyze patient clinical data and determine which ancillary diagnostic tests the patient qualifies for.

IMPORTANT RULES:
1. Be EXTREMELY LENIENT in qualifying patients. If there is ANY possible connection between the patient's conditions/symptoms/medications and a test, qualify them.
2. Err heavily on the side of qualifying. Even tangential or indirect connections count.
3. Common conditions like hypertension, diabetes, obesity, hyperlipidemia, anxiety, depression, pain, or being on multiple medications should qualify for MOST tests.
4. Age over 40 with ANY chronic condition qualifies for cardiovascular tests.
5. Any patient on psychiatric medications or with mood disorders qualifies for BrainWave and PGx.
6. Any patient with cardiovascular risk factors (HTN, DM, HLD, smoking, obesity, family history) qualifies for Carotid US, Echo, VitalWave.
7. Leg pain, swelling, or edema qualifies for lower extremity ultrasounds.
8. Multiple medications qualifies for PGx.

For each ancillary test, provide:
- qualifies: boolean (true if ANY connection exists - be very generous)
- confidence: "high" | "medium" | "low" (how strong the clinical indication is)
- for_clinician: A detailed, technical, evidence-based explanation citing the patient's specific conditions/medications from the provided data. Include clinical indications and explain the diagnostic value of the test for this patient's clinical picture. Reference specific comorbidities and how they interact to increase risk. Do NOT include any ICD-10 codes in this text. 4-5 sentences.
- for_patient: A warm, detailed explanation a non-clinical outreach caller can read to the patient on the phone explaining why their doctor recommends this test. Use their specific conditions in plain language, explain what the test looks for and why it matters for their health. Be reassuring and informative. Do NOT include any ICD-10 codes or medical codes in this text. 4-5 sentences. Start with "Based on..." or "Your doctor noticed..."
- qualifying_factors: Array of specific conditions/symptoms/meds from the patient's data that support qualification
- icd10_codes: Array of relevant ICD-10 codes that support the qualification

Available ancillary tests:
${ancillaryList.map(a => `- ${a.name}: ${a.desc}`).join("\n")}

Return JSON format:
{
  "qualifications": [
    {
      "ancillary": "test name",
      "qualifies": true/false,
      "confidence": "high/medium/low",
      "for_clinician": "...",
      "for_patient": "...",
      "qualifying_factors": ["factor1", "factor2"],
      "icd10_codes": ["E11.9", "I10"]
    }
  ]
}

Return ALL ${ancillaryList.length} tests in the response, ordered by qualification confidence (qualifying tests first).`
        },
        {
          role: "user",
          content: `Analyze this patient and qualify them for ancillary tests. Be VERY LENIENT - try to qualify for as many tests as possible.

Patient: ${patient_name || "Unknown"}
Age: ${age || "Unknown"} | Gender: ${gender || "Unknown"}

DIAGNOSES (Dx):
${dx || "Not provided"}

HISTORY/NOTES (Hx):
${hx || "Not provided"}

MEDICATIONS (Rx):
${rx || "Not provided"}`
        }
      ],
      temperature: 0.2,
      response_format: { type: "json_object" },
    });

    const parsed = JSON.parse(response.choices[0]?.message?.content || '{"qualifications":[]}');
    
    // Sort: qualifying first, then by confidence
    const confOrder = { high: 0, medium: 1, low: 2 };
    if (parsed.qualifications) {
      parsed.qualifications.sort((a, b) => {
        if (a.qualifies && !b.qualifies) return -1;
        if (!a.qualifies && b.qualifies) return 1;
        return (confOrder[a.confidence] || 3) - (confOrder[b.confidence] || 3);
      });
    }

    res.json({ ok: true, data: parsed });
  } catch (error) {
    console.error("[ai] Ancillary qualification failed:", error);
    res.status(500).json({ ok: false, error: "Failed to analyze qualifications" });
  }
});
