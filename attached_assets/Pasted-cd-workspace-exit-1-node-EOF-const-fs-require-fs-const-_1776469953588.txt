const fs = require("fs");
const path = require("path");

function ensureDir(filePath) {
  fs.mkdirSync(path.dirname(filePath), { recursive: true });
}

function writeFile(filePath, content) {
  ensureDir(filePath);
  fs.writeFileSync(filePath, content);
  console.log("✓ wrote", filePath);
}

function patchFile(filePath, transform) {
  const original = fs.readFileSync(filePath, "utf8");
  const updated = transform(original);
  if (updated === original) {
    console.log("• no changes needed", filePath);
    return;
  }
  fs.writeFileSync(filePath, updated);
  console.log("✓ patched", filePath);
}

type ScheduleBatch = {
  id: number;
  facility: string | null;
  scheduleDate: string | null;
  clinicianName: string | null;

type SchedulePatient = {
  id: number;

  insurance?: string | null;

  appointmentStatus?: string | null;
  patientType?: string | null;

function s(v: unknown): string {
  return v == null ? "" : String(v).trim();
}

function canonicalDay(v?: string | null): string {
  return s(v).slice(0, 10);
}

function getCoverageOwner(batch: ScheduleBatch): string {
  return s(batch.facility) || "Unassigned Facility";
}

function getProviderName(batch: ScheduleBatch): string {
  return s(batch.clinicianName) || "No provider";
}

export async function buildOutreachDashboard(storage: IStorage, today: string) {
  const batches = await storage.getAllScreeningBatches();
  const map = new Map<string, any>();

  for (const batch of batches as ScheduleBatch[]) {
    const batchDay = canonicalDay(batch.scheduleDate);
    if (batchDay && batchDay !== today) continue;

    const patients = await storage.getPatientScreeningsByBatch(batch.id);
    const owner = getCoverageOwner(batch);
    const ownerId = owner.toLowerCase().replace(/\\s+/g, "-");
    const providerName = getProviderName(batch);

    if (!map.has(ownerId)) {
      map.set(ownerId, {
        id: ownerId,
        name: owner,
        totalPatients: 0,
        touchedCount: 0,
        scheduledCount: 0,
        pendingCount: 0,
        conversionRate: 0,
        callList: [],
      });
    }

    const entry = map.get(ownerId);

    for (const patient of patients as SchedulePatient[]) {
      const appointmentStatus = s(patient.appointmentStatus) || "pending";
      const normalizedStatus = appointmentStatus.toLowerCase();
      const isTouched = normalizedStatus !== "pending";
      const isScheduled =
        normalizedStatus.includes("scheduled") ||
        normalizedStatus.includes("booked");

      entry.totalPatients += 1;
      if (isTouched) entry.touchedCount += 1;
      if (isScheduled) entry.scheduledCount += 1;
      if (!isTouched) entry.pendingCount += 1;

      entry.callList.push({
        id: \`\${batch.id}-\${patient.id}\`,
        patientId: patient.id,
        patientName: patient.name,
        facility: owner,
        phoneNumber: s(patient.phoneNumber) || "No phone",
        insurance: s(patient.insurance) || "No insurance",
        qualifyingTests: Array.isArray(patient.qualifyingTests)
          ? patient.qualifyingTests
          : [],
        appointmentStatus,
        patientType: s(patient.patientType) || "visit",
        batchId: batch.id,
        scheduleDate: s(batch.scheduleDate),
        time: s(patient.time) || "No time",
        providerName,
      });
    }
  }

  const coverageCards = Array.from(map.values())
    .map((entry: any) => ({
      ...entry,
      callList: entry.callList.sort(
        (a: any, b: any) =>
          a.time.localeCompare(b.time) ||
          a.patientName.localeCompare(b.patientName),
      ),
      conversionRate:
        entry.totalPatients > 0
          ? Math.round((entry.scheduledCount / entry.totalPatients) * 100)
          : 0,
    }))
    .sort(
      (a: any, b: any) =>
        b.totalPatients - a.totalPatients || a.name.localeCompare(b.name),
    );

  const totalPatients = coverageCards.reduce(
    (sum: number, card: any) => sum + card.totalPatients,
    0,
  );
  const totalScheduled = coverageCards.reduce(
    (sum: number, card: any) => sum + card.scheduledCount,
    0,
  );

  return {
    today,
    metrics: {
      coverageCount: coverageCards.length,
      totalCalls: coverageCards.reduce(
        (sum: number, card: any) => sum + card.touchedCount,
        0,
      ),
      totalScheduled,
      totalPending: coverageCards.reduce(
        (sum: number, card: any) => sum + card.pendingCount,
        0,
      ),
      avgConversion:
        totalPatients > 0
          ? Math.round((totalScheduled / totalPatients) * 100)
          : 0,
    },
    coverageCards,
  };
}
`);

function statusBadgeClass(status?: string | null) {
  const n = String(status || "pending").toLowerCase();

function formatDisplayDate(value?: string | null) {
  const raw = String(value || "").trim();
  if (!raw) return "Unscheduled";
  if (/^\\d{4}-\\d{2}-\\d{2}$/.test(raw)) {
    const [y, m, d] = raw.split("-").map(Number);
    const dt = new Date(y, (m || 1) - 1, d || 1);
    return dt.toLocaleDateString(undefined, { month: "short", day: "numeric", year: "numeric" });
  }
  const dt = new Date(raw);
  if (Number.isNaN(dt.getTime())) return raw;
  return dt.toLocaleDateString(undefined, { month: "short", day: "numeric", year: "numeric" });
}
