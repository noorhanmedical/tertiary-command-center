function sameDayString(value?: string | null) {
  if (!value) return "";
  const trimmed = String(value).trim();
  if (!trimmed) return "";
  const d = new Date(trimmed);
  if (Number.isNaN(d.getTime())) return trimmed;
  return d.toISOString().slice(0, 10);
}

function formatDisplayDate(value?: string | null) {
  if (!value) return "Unscheduled";
  const d = new Date(value);
  if (Number.isNaN(d.getTime())) return String(value);
  return d.toLocaleDateString(undefined, {
    month: "short",
    day: "numeric",
    year: "numeric",
  });
}

function formatDisplayTime(value?: string | null) {
  const raw = String(value || "").trim();
  return raw || "No time";
}

function statusBadgeClass(status?: string | null) {
  const normalized = String(status || "pending").toLowerCase();
  if (normalized.includes("scheduled") || normalized.includes("booked")) {
    return "bg-green-100 text-green-700 border-green-200";
  }
  if (normalized.includes("complete")) {
    return "bg-blue-100 text-blue-700 border-blue-200";
  }
  if (normalized.includes("cancel")) {
    return "bg-red-100 text-red-700 border-red-200";
  }
  return "bg-amber-100 text-amber-700 border-amber-200";
}

function cardShellClass(active = false) {
  return [

  ].join(" ");
}

    queryFn: async () => {
      const batchesRes = await fetch("/api/screening-batches");
      if (!batchesRes.ok) throw new Error("Failed to load schedules");
      const batches: ScreeningBatch[] = await batchesRes.json();

      const details = await Promise.all(
        batches.map(async (batch) => {
          const res = await fetch(`/api/screening-batches/${batch.id}`);
          if (!res.ok) return { ...batch, patients: [] as PatientScreening[] };
          return res.json();
        }),
      );

      return details as BatchWithPatients[];

    refetchInterval: 60000,
  });

  const summaries = useMemo<TeamMemberSummary[]>(() => {
    const today = new Date().toISOString().slice(0, 10);
    const relevantBatches = (data || []).filter((batch) => {
      const batchDay = sameDayString(batch.scheduleDate);
      return batchDay === today || !batchDay;
    });

    const map = new Map<string, TeamMemberSummary>();

    for (const batch of relevantBatches) {
      const teamMemberName = String(batch.clinicianName || "").trim() || "Unassigned Team Member";
      const key = teamMemberName.toLowerCase();

      if (!map.has(key)) {
        map.set(key, {
          id: key,
          name: teamMemberName,
          totalPatients: 0,
          touchedCount: 0,
          scheduledCount: 0,
          pendingCount: 0,
          conversionRate: 0,
          facilities: [],
          callList: [],
        });
      }

      const entry = map.get(key)!;
      const facility = String(batch.facility || "").trim() || "Unassigned Facility";

      if (!entry.facilities.includes(facility)) {
        entry.facilities.push(facility);
      }

        const normalizedStatus = appointmentStatus.toLowerCase();
        const isTouched = normalizedStatus !== "pending";
        const isScheduled = normalizedStatus.includes("scheduled") || normalizedStatus.includes("booked");

        entry.totalPatients += 1;
        if (isTouched) entry.touchedCount += 1;
        if (isScheduled) entry.scheduledCount += 1;
        if (!isTouched) entry.pendingCount += 1;

        entry.callList.push({
          id: `${batch.id}-${patient.id}`,
          patientId: patient.id,
          patientName: patient.name,
          time: formatDisplayTime(patient.time),
          facility,
          phoneNumber: String(patient.phoneNumber || "").trim() || "No phone",
